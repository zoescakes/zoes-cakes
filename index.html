db.collection("orders").onSnapshot(snapshot => { const ordersDiv = document.getElementById("orders"); ordersDiv.innerHTML = ""; let paidDaily = 0; let paidMonthly = 0; let claimedDaily = 0; let claimedMonthly = 0; const today = new Date().toISOString().slice(0,10); const thisMonth = today.slice(0,7); snapshot.forEach(doc => { const o = doc.data(); if (o.status === "Paid") { if (o.date === today) paidDaily += Number(o.price); if (o.date.startsWith(thisMonth)) paidMonthly += Number(o.price); } if (o.status === "Claimed") { if (o.date === today) claimedDaily += Number(o.price); if (o.date.startsWith(thisMonth)) claimedMonthly += Number(o.price); } ordersDiv.innerHTML += `
${o.orderNo || ""}
👤 ${o.name}
🎂 ${o.cake}
📏 ${o.size}
💵 ₱${o.price}
📅 ${o.date}
💖 ${o.status}

`; }); document.getElementById("paidDaily").innerText = paidDaily; document.getElementById("paidMonthly").innerText = paidMonthly; document.getElementById("claimedDaily").innerText = claimedDaily; document.getElementById("claimedMonthly").innerText = claimedMonthly; });
